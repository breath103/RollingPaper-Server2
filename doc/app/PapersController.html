def show 
  @paper = Paper.find_by_id(params[:id])
  if(@paper && @paper.state != "opened")
    @paper.send_feedback_to_participants "이나연 님이 #{@paper.title}를 보셨습니다"
    @paper.state = "opened"
    @paper.save!
  end
  render layout: false
end
